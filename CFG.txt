P → D; P | D
D → fn ID (A) : ID { STMTS }
A → ID, A | ε
ID → [a-z | A-Z | _][a-z | A-Z | _ | 0-9 ]*
STMTS → ST; STMTS | ε
ST → ID := E
  | if BOOL { STMTS } else { STMTS } 
  | while BOOL { STMTS }
  | for INT : INT { STMTS }
E → INT
  | ID
  | E + E 
  | E - E 
  | E * E
  | E / E
  | E % E
  | ID(EA)
BOOL → E COMP E
  | ! E COMP E
COMP → =
  | <
  | >
  | <=
  | >=
EA → ε | E, EA
INT → [0-9]+
